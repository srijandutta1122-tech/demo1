#include <iostream>
using namespace std;

int main()
{
    int p[] = {0, 1, 4, 5, 7};
    int wt[] = {0, 1, 3, 4, 5};
    int n = 4;
    int m = 7;

    int k[n+1][m+1];

    for(int i = 0; i <= n; i++)
    {
        for(int w = 0; w <= m; w++)
        {
            if (i == 0 || w == 0)
            {
                k[i][w] = 0;
            }
            else if(wt[i] <= w)
            {
                k[i][w] = max(p[i] + k[i-1][w-wt[i]], k[i-1][w]);
            }
            else
            {
                k[i][w] = k[i-1][w];
            }
        }
    }

    cout<<"Maximum Profit: "<<k[n][m]<<endl;

    cout<<"Items included in the knapsack:"<<endl;
    int i = n, j = m;
    while(i > 0 && j > 0)
    {
        if(k[i][j] == k[i-1][j])
        {
            cout<<"Item "<<i<<": Not included"<< endl;
            i--;
        }
        else
        {
            cout<<"Item "<<i<<": Included"<<endl;
            j = j - wt[i];
            i--;
        }
    }

    return 0;
}
